\section{Introduction}
\label{sec:intro}

\subsection{About}
\label{subsec:about}

This package provides \cs{clock} and \cs{calendardate}.
\cs{clock} prints the time of day.
\cs{calendardate} prints the calendar date.
These commands are ideal for text markup because their syntax is short,
flexible, easy-to-understand, and unambiguous.
The printed time of day and calendar date are highly customizable through a
variety of key-value options.
For advanced users, this package also provides robust \LaTeX3 programming
support.

\label{subsec:loading}

\listheading{Requirements:}
\begin{itemize}
  \item \LaTeXe{} version 2023-11-01 or newer
  \item \pkg{l3kernel} version 2023-11-01 or newer
\end{itemize}

\section{Commands}
\label{sec:cmds}

This section documents the commands provided by the \pkg{clockdate} package.

\begin{code}{\clockdatesetup}
  \begin{syntax}
    \cs{clockdatesetup} \marg{key-value list}
  \end{syntax}
  Sets and processes the \pkg{clockdate} package keys (\S\ref{sec:keys}) in
  \meta{key-value list}.
  All assignments made by \cs{clockdatesetup} are local to the current group.
  Can be used mid-document.
\end{code}

\begin{code}{\clock}
  \begin{syntax}
    \cs{clock} \sarg{} \oarg{key-value list} \marg{token list}
  \end{syntax}
  Prints the time of day or time-of-day range specified by \meta{token list}.
  \meta{token list} must take either of the following forms:
  \begin{itemize}
    \item \meta{clock}
    \item \meta{clock_1}|--|\meta{clock_2}
  \end{itemize}

  \meta{clock} must take any of the following forms:
  \begin{itemize}
    \item \meta{hour}
    \item \meta{hour}|:|\meta{minute}
    \item \meta{hour}|:|\meta{minute}|:|\meta{second}
  \end{itemize}

  \meta{hour} must be an \meta{integer} from \num{0} to \num{24}.
  \meta{minute} must be an \meta{integer} from \num{0} to \num{59}.
  \meta{second} must be an \meta{integer} from \num{0} to \num{60}.
  (The value of \num{60} allows for leap seconds.)
  If $\meta{hour}=\num{24}$, then \meta{minute} and \meta{second} must be zero.

  Adding the optional argument sets the keys in \meta{key-value list} in path
  |clock| for only that instance of \cs{clock}.
  For details on adding the optional star, see the key \key{clock/star}
  (\S\ref{subsec:clock}).
\end{code}

\begin{code}{\calendardate}
  \begin{syntax}
    \cs{calendardate} \sarg{} \oarg{key-value list} \marg{token list}
  \end{syntax}
  Prints the calendar date specified by \meta{token list}.
  \meta{token list} must take any of the following forms:
  \begin{itemize}
    \item \meta{year}|-|\meta{month}|-|\meta{day(s)}
    \item \meta{month}|-|\meta{day(s)}
    \item \meta{year}|-|\meta{month(s)}
    \item \meta{year}
  \end{itemize}

  \meta{year} and \meta{day} must be an \meta{integer}.
  \meta{year} may be substituted with \meta{empty}|-|\meta{year} to ensure that
  the year is always before the common era.
  \meta{month} must be an \meta{integer} from \num{1} to \num{12}.
  The calendar date must be valid.
  \date{2-29} is always valid if \meta{year} is omitted.
  If \meta{year} is provided, \date{2-29} is error-checked using the Julian
  calendar.

  \meta{day(s)} must take any of the following forms:
  \begin{itemize}
    \item a single day \meta{day}
    \item a day range \meta{day_1}|--|\meta{day_2}
    \item
    a comma-separated list where each item must be either a single day
    \meta{day} or a day range \meta{day_1}|--|\meta{day_2}
  \end{itemize}

  \meta{month(s)} must take any of the following forms:
  \begin{itemize}
    \item a single month \meta{month}
    \item a month range \meta{month_1}|--|\meta{month_2}
    \item
    a comma-separated list where each item must be either a single month
    \meta{month} or a month range \meta{month_1}|--|\meta{month_2}
  \end{itemize}

  \meta{token list} must not contain |--| at the top level unless it is used to
  delimit a day or month range.
  Adding the optional argument sets the keys in \meta{key-value list} in path
  |date| for only that instance of \cs{calendardate}.
  For details on adding the optional star, see the key \key{date/star}
  (\S\ref{subsec:date}).
\end{code}


\section{Keys}
\label{sec:keys}

This section documents the keys provided by the \pkg{clockdate} package.
Set the package keys using \cs{clockdatesetup} (\S\ref{sec:cmds}).

\begin{code}{overwrite-date, \date}
  Redefines \cs{date} as equivalent to \cs{calendardate} (\S\ref{sec:cmds}).
  This is done at the \hook{begindocument} hook so that the original definition
  remains usuable in the preamble.

\end{code}

  Meta key that sets the keys \key{clock/reset} and \key{date/reset}.
\end{code}

\begin{code}{clock/star}
  \begin{syntax}
    \key{clock/star=\marg{key-value list}}
  \end{syntax}

\end{code}

\begin{code}{clock/clock}
  \begin{syntax}
    \key{clock/clock=12-hour|24-hour}
  \end{syntax}
  Choice key that sets whether to print the hour using the 12-hour or 24-hour
  clock.
  If the value is |24-hour|, the am/pm is always omitted.
  The initial value is |12-hour|.
\end{code}

\begin{code}{clock/range-separator}
  \begin{syntax}
    \key{clock/range-separator=\meta{token list}}
  \end{syntax}
  Sets the separator between \meta{clock_1} and \meta{clock_2} in a time-of-day
  range to \meta{token list}.
  The initial value is | to |.
\end{code}

\subsubsection{Printing the am/pm}
\label{subsubsec:printampm}

\begin{code}{clock/ampm}
  \begin{syntax}
    \key{clock/ampm=a.m.|am|a|A.M.|AM|A|none}
  \end{syntax}
  Choice key that sets the format in which the am/pm is printed.
  The initial value is |a.m.|
  \input{clockdate-input-ampm.tex}

  The am/pm is omitted if \key{clock/ampm=none} or \key{clock/clock=24-hour}.

  If the value is |a.m.| or |A.M.|, the trailing am/pm will not print an extra
  period if the next token after \cs{clock} is a period.
  If the am/pm prints the period, it is followed by \cs{@} because the sentence
  does not end here.
\end{code}

  \end{syntax}
  Boolean key that sets whether \meta{clock_1} prints the am/pm in a
  time-of-day range if \meta{clock_1} and \meta{clock_2} either both print am
  or both print pm.
  The initial value is |true|.
\end{code}

\begin{code}{date/priority}
  \begin{syntax}
    \key{date/priority=month-day|year-month}
  \end{syntax}

  \cs{calendardate} to differentiate between typesetting
  \meta{month}|-|\meta{day(s)} and \meta{year}|-|\meta{month(s)}.
  This situation occurs if \meta{token list} contains exactly two fields and
  all values within are an \meta{integer} from \num{1} to \num{12}.
  This key sets the prioritized representation.
  The initial value is |month-day|.
\end{code}

\begin{code}{date/order}
  \begin{syntax}
    \key{date/order=day-month-year|year-month-day|month-day-year}
  \end{syntax}

  The initial value is |day-month-year|.
\end{code}

\subsubsection{Printing the year, month, and day}
\label{subsubsec:printymd}

\begin{code}{date/year}
  \begin{syntax}
    \key{date/year=y|yy|yyyy}
  \end{syntax}
  Choice key that sets the format in which the year is printed.
  |y| prints the year with the minimum number of digits.
  |yy| prints the year with only the last two digits.
  |yyyy| prints the year with a minimum of four digits, adding a leading zero
  if necessary.
  The initial value is |y|.
\end{code}

\begin{code}{date/month}
  \begin{syntax}
    \key{date/month=m|mm|mmm|mmmm}
  \end{syntax}
  Choice key that sets the format in which the month is printed.
  |m| prints the month with the minimum number of digits.
  |mm| prints the month with two digits, adding a leading zero if necessary.
  |mmm| prints the month in short form.
  |mmmm| prints the month with the name of the month fully spelled out.
  The initial value is |mmmm|.

  By default, |mmm| prints the month in the same form as |mmmm|, except that
  some months are abbreviated: Jan., Feb., Aug., Sept., Oct., Nov., Dec.
  The abbreviated month will not print an extra period if the next token after
  \cs{calendardate} is a period and \cs{calendardate} ends with an abbreviated
  month that ends in a period.
  If the abbreviated month prints the period, it is followed by \cs{@} because
  the sentence does not end here.
\end{code}

\begin{code}{date/day}
  \begin{syntax}
    \key{date/day=d|dd}
  \end{syntax}
  Choice key that sets the format in which the day is printed.
  |d| prints the day with the minimum number of digits.
  |dd| prints the day with two digits, adding a leading zero if necessary.
  The initial value is |d|.
\end{code}

\begin{code}{date/day-ordinal}
  \begin{syntax}
    \key{date/day-ordinal=true|false}
  \end{syntax}
  Boolean key that sets whether to print the day as an ordinal day
  (e.g. 1\textsuperscript{st}, 4\textsuperscript{th}).
  The initial value is |false|.
\end{code}

  The value of \meta{hour_2}, \meta{minute_2}, and \meta{second_2} in a
  time-of-day range.
  The value is \num{-1} if omitted.
\end{code}

\begin{code}{\l_clockdate_clock_ampm_print_bool}
  Boolean that stores whether the am/pm will be printed.
  The value is |true| if \key{clock/clock=12-hour} (\S\ref{subsec:clock}) and
  \key[neq]{clock/ampm=none} (\S\ref{subsubsec:printampm}).

\begin{code}{\l_clockdate_clock_ampm_print_first_bool}

\subsection{Variables set in \cs{calendardate}}
\label{subsec:vardate}

This subsection documents the variables that are set within \cs{calendardate}.
Outside of \cs{calendardate}, they do not have defined behavior.

\begin{code}{\l_clockdate_date_year_str}
  The value of \meta{year}.
  The value is \meta{empty} if \meta{year} is omitted.
\end{code}

\begin{code}{\l_clockdate_date_month_str}
  The value of \meta{month(s)}.
  The value takes the same form as the \meta{month(s)} as explained for
  \cs{calendardate}.
  The value is an \meta{integer} if exactly one month is printed.
  The value is \meta{empty} if no months are printed.
\end{code}

\begin{code}{\l_clockdate_date_day_str}
  The value of \meta{day(s)}.
  The value takes the same form as the \meta{day(s)} as explained for
  \cs{calendardate}.
  The value is an \meta{integer} if exactly one day is printed.
  The value is \meta{empty} if \meta{day(s)} is omitted.
\end{code}

\subsection{Printing the months}

\begin{code}

  Stores the months used if the value of the key \key{date/month} is |mmmm| and
  |mmm|, respectively.
  The initial values are displayed in in Table \ref{table:months-seq}.
\end{code}

\input{clockdate-input-months-seq.tex}

\begin{code}{\clockdate_end_period:}
  The purpose of this function is to avoid mistakenly printing two consecutive
  periods if the month is abbreviated and ends with a period.
  Prints a period unless all of the following is true:
  \begin{itemize}
    \item The display of \cs{calendardate} ends with the month.
    \item The month ends with \cs{clockdate_end_period:}.
    \item The next token after \cs{calendardate} is a period.
  \end{itemize}
  If \cs{clockdate_end_period:} prints the period, it is followed by \cs{@}
  because the sentence does not end here.
\end{code}

\subsection{Printing the ordinal day}

\begin{code}{\clockdate_date_day_ordinal_format:n}
  \begin{syntax}
    \cs{clockdate_date_day_ordinal_format:n} \marg{day}
  \end{syntax}
  Sets the format in which the suffix of the ordinal day is printed.
  |#1| represents the value of \meta{day}.
  The initial definition is:
  \begin{verbatim}
\cs_new_protected:Npn \clockdate_date_day_ordinal_format:n #1
  {
    \textsuperscript
      {
        \int_case:nnF {#1}
          {
            {  1 } { st } {  2 } { nd } {  3 } { rd }
            { 21 } { st } { 22 } { nd } { 23 } { rd }
            { 31 } { st }
          }
          { th }
      }
  }
  \end{verbatim}
\end{code}
