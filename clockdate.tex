\section*{\LARGE The \pkg{clockdate} package}

Print the time of day and the calendar date.

Oliver Beery

Version 0.0.0\quad\date{2026-03-05}


\section{Introduction}
\label{sec:introduction}

\subsection{About}

This package provides \cs{clock} and \cs{calendardate}.
\cs{clock} prints the time of day.
\cs{calendardate} prints the calendar date.

\subsection{Loading the package}
\label{subsec:loading}

\listheading{Requirements:}
\begin{itemize}
  \item \LaTeXe{} version 2023-11-01 or newer
  \item \pkg{l3kernel} version 2023-11-01 or newer
\end{itemize}


\section{Package option}
\label{sec:packageoption}

\begin{code}{overwrite-date, \date}
  Redefines \cs{date} as equivalent to \cs{calendardate}.
  This is done at the \hook{begindocumnt} hook so that the original definition
  remains usuable in the preamble.
\end{code}


\section{Commands}
\label{sec:commands}

This section documents the commands provided by the \pkg{clockdate} package.

\begin{code}{\clockdatesetup}
  \begin{syntax}
    \cs{clockdatesetup} \sarg{} \marg{key-value list}
  \end{syntax}
  Sets and processes the \pkg{clockdate} package keys (\S\ref{sec:keys}) in
  \meta{key-value list}.
  Adding the optional star first sets all the \pkg{clockdate} package keys to
  their initial values.
  All assignments made by \cs{clockdatesetup} are local to the current group.
  Can be used mid-document.
\end{code}

\begin{code}{\clock}
  \begin{syntax}
    \cs{clock} \sarg{} \oarg{key-value list} \marg{token list}
  \end{syntax}
  Prints the time of day or a time-of-day range.
  \meta{token list} must take either of the following forms:
  \begin{itemize}
    \item \meta{clock}
    \item \meta{clock_1}|--|\meta{clock_2}
  \end{itemize}

  \meta{clock} must take any of the following forms:
  \begin{itemize}
    \item \meta{hour}
    \item \meta{hour}|:|\meta{minute}
    \item \meta{hour}|:|\meta{minute}|:|\meta{second}
  \end{itemize}

  \meta{hour} must be an \meta{integer} from \num{0} to \num{24}.
  \meta{minute} must be an \meta{integer} from \num{0} to \num{59}.
  \meta{second} must be an \meta{integer} from \num{0} to \num{60}.
  (The value of \num{60} allows for leap seconds.)

  Adding the optional argument sets the keys in \meta{key-value list} in path
  |clock| for only that particular usage of \cs{clock}.
  For details on adding the optional star, see the key \key{clock/star}
  (\S\ref{subsec:clock}).
\end{code}

\begin{code}{\calendardate}
  \begin{syntax}
    \cs{calendardate} \sarg{} \oarg{key-value list} \marg{token list}
  \end{syntax}
  \listheading
    {
      Prints the calendar date.
      \meta{token list} must take any of the following forms:
    }
  \begin{itemize}
    \item \meta{year}|-|\meta{month}|-|\meta{day(s)}
    \item \meta{month}|-|\meta{day(s)}
    \item \meta{year}|-|\meta{month(s)}
  \end{itemize}

  \meta{year} must be an \meta{integer} from \num{1000} to \num{9999}.
  \meta{month} must be an \meta{integer} from \num{1} to \num{12}.

  \listheading{\meta{day(s)} must take any of the following forms:}
  \begin{itemize}
    \item a single day \meta{day}
    \item a day range \meta{day_1}|--|\meta{day_2}
    \item a comma-separated list of items where each item must be either a
    single day \meta{day} or a day range \meta{day_1}|--|\meta{day_2}
  \end{itemize}
  \meta{day} must be an \meta{integer}.
  The calendar date must be valid.
  \date{2-29} is always valid if \meta{year} is omitted.

  \listheading{\meta{month(s)} must take any of the following forms:}
  \begin{itemize}
    \item a single month \meta{month}
    \item a month range \meta{month_1}|--|\meta{month_2}
    \item a comma-separated list of items where each item must be either a
    single month \meta{month} or a month range \meta{month_1}|--|\meta{month_2}
  \end{itemize}

  Adding the optional argument sets the keys in \meta{key-value list} in path
  |date| for only that particular usage of \cs{calendardate}.
  For details on adding the optional star, see the key \key{date/star}
  (\S\ref{subsec:date}).
\end{code}

\subsection{\cs{clock}}
\label{subsec:clock}

This subsection documents the keys that modify the behavior of \cs{clock}.

\begin{code}{clock}
  \begin{syntax}
    \key{clock=\marg{key-value list}}
  \end{syntax}
  Meta key that sets the keys in \meta{key-value list} in path |clock|.
\end{code}

\begin{code}{clock/star}
  \begin{syntax}
    \key{clock/star=\marg{key-value list}}
  \end{syntax}
  When adding the optional star in \cs{clock}, the keys in \meta{key-value
  list} in path |clock| will be set for only that particular usage of
  \cs{clock}.
  The initial value is \meta{empty}.
\end{code}

\begin{code}{clock/format}
  \begin{syntax}
    \key{clock/format=\choices{12-hour,24-hour}}
  \end{syntax}
  Choice key that sets whether to print the hour in 12-hour or 24-hour format.
  In 24-hour format, the am/pm is always omitted.
  The initial value is \key{=12-hour}.
\end{code}

\begin{code}{clock/range-separator}
  \begin{syntax}
    \key{clock/range-separator=\meta{token list}}
  \end{syntax}
  Sets the separator between \meta{clock_1} and \meta{clock_2} in a time-of-day
  range to \meta{token list}.
  The initial value is \key{= to }.
\end{code}

\subsubsection{Printing the hour, minute, and second}

\begin{code}{clock/hour}
  \begin{syntax}
    \key{clock/hour=\choices{h,hh}}
  \end{syntax}
  Choice key that sets the format in which the hour is printed.
  \key{=h} prints the hour with the minimum number of digits.
  \key{=hh} prints the hour with two digits, adding a leading zero if
  necessary.
  The initial value is \key{=h}.
\end{code}

\begin{code}{clock/minute}
  \begin{syntax}
    \key{clock/minute=\choices{m,mm}}
  \end{syntax}
  Choice key that sets the format in which the minute is printed.
  \key{=m} prints the minute with the minimum number of digits.
  \key{=mm} prints the minute with two digits, adding a leading zero if
  necessary.
  The initial value is \key{=mm}.
\end{code}

\begin{code}{clock/second}
  \begin{syntax}
    \key{clock/second=\choices{s,ss}}
  \end{syntax}
  Choice key that sets the format in which the second is printed.
  \key{=s} prints the second with the minimum number of digits.
  \key{=ss} prints the second with two digits, adding a leading zero if
  necessary.
  The initial value is \key{=ss}.
\end{code}

\subsubsection{Separating the hour, minute, and second}

\begin{code}{clock/hour-minute-separator}
  \begin{syntax}
    \key{clock/hour-minute-separator=\meta{token list}}
  \end{syntax}
  Sets the separator between \meta{hour} and \meta{minute} to
  \meta{token list}.
  The initial value is \key{=:}.
\end{code}

\begin{code}{clock/minute-second-separator}
  \begin{syntax}
    \key{clock/minute-second-separator=\meta{token list}}
  \end{syntax}
  Sets the separator between \meta{minute} and \meta{second} to
  \meta{token list}.
  The initial value is \key{=:}.
\end{code}

\begin{code}{clock/clock-separator}
  \begin{syntax}
    \key{clock/clock-separator=\meta{token list}}
  \end{syntax}
  Meta key that sets the keys \key{clock/hour-minute-separator} and
  \key{clock/minute-second-separator} to \meta{token list}.
\end{code}

\subsubsection{Printing the am/pm}

\begin{code}{clock/ampm}
  \begin{syntax}
    \key{clock/ampm=\choices{a.m.,am,a,A.M.,AM,A,none}}
  \end{syntax}
  Choice key that sets the format in which the am/pm is printed.

\begin{code}{clock/ampm-separator}
  \begin{syntax}

  \end{syntax}
  Sets the separator before the am/pm to \meta{token list}.
  The initial value is \key{= }.
\end{code}

\begin{code}{clock/ampm-omit-first}
  \begin{syntax}
    \key{clock/ampm-omit-first=\choices{true,false}}
  \end{syntax}
  Boolean key that sets whether \meta{clock_1} prints the am/pm in a
  time-of-day range if \meta{clock_1} and \meta{clock_2} would either both
  print am or both print pm.
  The initial value is \key{=true}.
\end{code}

\subsubsection{Printing the year, month, and day}

\begin{code}{date/year}
  \begin{syntax}
    \key{date/year=\choices{yy,yyyy}}
  \end{syntax}
  Choice key that sets the format in which the year is printed.
  \key{=yy} prints the year with only the last two digits.
  \key{=yyyy} prints the 4-digit year.
  The initial value is \key{=yyyy}.
\end{code}

\begin{code}{date/month}
  \begin{syntax}
    \key{date/month=\choices{m,mm,mmm,mmmm}}
  \end{syntax}
  Choice key that sets the format in which the month is printed.
  \key{=m} prints the month with the minimum number of digits.
  \key{=mm} prints the month with two digits, adding a leading zero if
  necessary.
  \key{=mmm} prints the month in short form.
  \key{=mmmm} prints the month with the name of the month fully spelled out.
  By default, \key{=mmm} prints the month in the same form as \key{=mmmm},
  except that some months are abbreviated: Jan., Feb., Aug., Sept., Oct.,
  Nov., Dec.
  The initial value is \key{=mmmm}.

  % If the next token after \cs{calendardate} is a period, the abbreviated month
  % will not print an extra period.
  % When the abbreviated month prints the period, it is followed by \cs{@}
  % because the sentence does not end here.
\end{code}

\begin{code}{date/day}
  \begin{syntax}
    \key{date/day=\choices{d,dd}}
  \end{syntax}
  Choice key that sets the format in which the day is printed.
  \key{=d} prints the day with the minimum number of digits.
  \key{=dd} prints the day with two digits, adding a leading zero if necessary.
  The initial value is \key{=d}.
\end{code}

\begin{code}{\clockdate_date_month_period:}
  % This command should be used only in \cs{l_clockdate_date_months_mmmm_seq} and
  % \cs{l_clockdate_date_months_mmm_seq}.
  The purpose of this command is to avoid mistakenly printing two consecutive
  periods if the month is abbreviated.
  Prints a period unless all of the following is true:

  \begin{itemize}
    \item The display of \cs{calendardate} ends with the month.
    \item The month ends with \cs{clockdate_date_month_period:}.
    \item The next token after \cs{calendardate} is a period.
  \end{itemize}

  When \cs{clockdate_date_month_period:} prints the period, it is followed by
  \cs{@} because the sentence does not end here.
\end{code}

This package uses some ideas from the \pkg{datetime2} and \pkg{siunitx}
packages.

The \pkg{clockdate} keys on the left resemble the corresponding \pkg{datetime2}
option on the right.
\input{clockdate-input-datetime2.tex}

The \pkg{clockdate} package keys
\begin{itemize}
  \item \key{date/days-final-separator}
  \item \key{date/days-pair-separator}
  \item \key{date/days-separator}
  \item \key{date/months-final-separator}
  \item \key{date/months-pair-separator}
  \item \key{date/months-separator}
\end{itemize}
use a similar naming convention to the following \pkg{siunitx} options:
\begin{itemize}
  \item \key{list-final-separator}
  \item \key{list-pair-separator}
  \item \key{list-separator}
\end{itemize}
